showSession :: arg2:World -> arg1:User -> arg0:PaperId -> World
showSession = \w . \client .
    \p . 
      let row = do
                  t <- getPaperTitle w p
                  auts <- do
                            g0 <- downgrade (do
                                               x5 <- getPhase w
                                               return (eq Done x5))
                            g1 <- downgrade (do
                                               x10 <- getPaperAuthors w p
                                               return (elem client x10))
                            
                            if g0 || g1
                              then getPaperAuthors w p
                              else return Nil
                  st <- do
                          g2 <- downgrade (do
                                             x16 <- getPhase w
                                             return (eq Done x16))
                          
                          if g2
                            then getPaperStatus w p
                            else return NoDecision
                  
                  if st == Accepted
                    then do
                           ses <- getPaperSession w p
                           return (unwords (Cons t (Cons
                                                      (show auts) (Cons ses
                                                                     Nil))))
                    else do
                           ses <- return emptyString
                           return (unwords (Cons t (Cons
                                                      (show auts) (Cons ses
                                                                     Nil)))) in
      print w (return client) row
