      let row = do
                  t <- getPaperTitle w p
                  auts <- do
                            g0 <- downgrade (do
                                               x5 <- getPhase w
                                               return (eq Done x5))
                            
                            if g0
                              then getPaperAuthors w p
                              else return Nil
                  st <- do
                          g1 <- downgrade (do
                                             x11 <- getPhase w

                            then getPaperStatus w p
                            else return NoDecision
                  
                  if st == Accepted
                    then do
                           ses <- getPaperSession w p
                           return (unwords (Cons t (Cons
                                                      (show auts) (Cons ses
                                                                     Nil))))
                    else do
                           ses <- return emptyString
                           return (unwords (Cons t (Cons
                                                      (show auts) (Cons ses
